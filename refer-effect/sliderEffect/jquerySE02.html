const sliderWrap = $('.slider-wrap');
const sliderImg = $('.slider-img');
const sliderInner = $('.slider-inner');
const slider = $('.slider');

let currentIndex = 0;
let sliderCount = slider.length;
let sliderWidth = slider.width();
let sliderDuration = 600;

setInterval(function () {
    (currentIndex < sliderCount - 1) ? currentIndex++ : currentIndex = 0;
    let sliderPosition = currentIndex * -sliderWidth;
    sliderInner.animate({ left: sliderPosition }, sliderDuration, "easeOutExpo");
}, 3000);
