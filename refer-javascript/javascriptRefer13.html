101. 테이블 만들기 

{
        let num = 1;
        let table = "<table>";
        for (let i = 1; i <= 10; i++) {
            table += "<tr>";
            for (let j = 1; j <= 10; j++) {
                table += "<td>" + num + "</td>";
                num++;
            }
            table += "</tr>";
        }
        table += "</table>";
    
        document.write(table);
    }

102. 리버스 테이블 만들기 

{
        let num = 100;
        let table = "<table>";
        for (let i = 1; i <= 10; i++) {
            table += "<tr>";
            for (let j = 1; j <= 10; j++) {
                table += "<td>" + num + "</td>";
                num--;
            }
            table += "</tr>";
        }
        table += "</table>";
    
        document.write(table);
    }
